    "use strict";
this.name = "Additional Planets SR pack Redux";
this.author = "spara, Redspear";
this.copyright = "spara, Redspear";
this.description = "Texture pack for Additional Planets SR";
this.licence = "CC-BY-SA 3.0";
this.version = "1.1";

this.startUp = function() {
	if (!worldScripts["System Redux"]) return;
	var moons = new Array();
	for (var i = 1; i <= 8	; i++) moons.push("redux_moon" + i + ".png");
	worldScripts["System Redux"].addMoonsToPool(moons);
	var planets = new Array();
	for (var i = 1; i <= 16	; i++) planets.push("redux_planet" + i + ".png");
	worldScripts["System Redux"].addPlanetsToPool(planets);
}
