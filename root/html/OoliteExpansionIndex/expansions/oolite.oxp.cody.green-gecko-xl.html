this.description = "Script to very occasionly spawn a Green Gecko.";
this.version = "1.1";

this.startUp = function()
{
	this.evgg = false; // Will be set by shipscript in case the ship follows player
	this.evgg_system = -1; // Limit to max 1 occurence in a system
	missionVariables.EV_GG = null; // Get rid of it
	missionVariables.gg_Pilot = null; // Get rid of it
}
this.shipWillExitWitchspace = function()
{
	// Below statement will be executed only if a random number (range 0.0-1.0)
	// higher than 0.99 is returned, i.e. 1% of the time, no contact happened already
	// and if no GG is out there
	if (!this.evgg && this.evgg_system !== system.ID && Math.random() > 0.99){
		this.evgg = system.addShipsToRoute("ev_green_gecko", 1, 0.1)[0];
		this.evgg_system = system.ID;
	}
};
this.playerWillEnterWitchspace = function()
{
	this.evgg = false;
}
this.shipSpawned = function(ship)
{
   if(this.evgg && this.evgg.isValid && player.ship.isValid){
      if(this.evgg.position.distanceTo(player.ship.position) > 25600) return;
      var check = ["EQ_QC_MINE","EQ_RMB_CASCADE_MISSILE","EQ_RMB_LAW_MISSILE","EQ_RMB_OVERRIDE_MISSILE",
         "EQ_NUKE2_MISSILE","EQ_HARPOON_NUKE1_MISSILE","EQ_HARPOON_NUKE2_MISSILE","EQ_PHOTON_MISSILE",
         "RANDOM_HITS_MINE"];
      if(check.indexOf(ship.primaryRole) !== -1){
         player.consoleMessage("Attention: pylon-mounted weapon malfunction!",6);
         ship.remove(true);
      }
   }
}
this.shipWillDockWithStation = function(station)
{
	if(station.suppressArrivalReports) return; // Just in case
	if(missionVariables.evgg_PilotCount){
		var i = 1;
		while(i <= missionVariables.evgg_PilotCount){
			player.addMessageToArrivalReport("Kudos, Commander... you have proven worthy of your rank! For besting " + randomName(false) + " " + randomName(false) + ", one of the legendary " + expandDescription("[EVGG_FIXED_DESC]") + " of " + this.randomSystem() + ", you have been granted the princely sum of " + formatCredits(2000, true, true));
			player.credits += 2000;
			player.ship.manifest.slaves--;
			i +=1;
		}
		missionVariables.evgg_PilotCount = null;
	}
}
this.randomSystem = function()
{
	return System.systemNameForID(Math.floor(Math.random() * 255));
}

this.$scanForTarget = function()
{
	this.ship.target = player.ship;
	this.ship.reactToAIMessage("TARGET_FOUND");
};
this.shipDied = function(whom,why)
{
	this.ship.spawnOne("ev_green_gecko_escapePod");
	this.ship.commsMessage(expandDescription("[EVGG_KILLED_MESSAGE]"),player.ship);
}
this.GotAttacked = function()
{
	if(!this.firstMessage) return;
	if(!this.attacked){
		this.ship.commsMessage(expandDescription("[EVGG_ATTACKED_MESSAGE]"),player.ship);
		this.attacked = true;
	} else {
		if(this.hitTime>clock.absoluteSeconds) return;
		if(this.ship.energy>this.ship.maxEnergy*0.5) this.ship.commsMessage(expandDescription("[EVGG_ATTACKED_HIT]"),player.ship);
		else this.ship.commsMessage(expandDescription("[EVGG_ATTACKED_DAMAGED]"),player.ship);
	}
	this.hitTime = clock.absoluteSeconds+30;
	return;
}
this.firstAttack = function()
{
	if(!this.firstMessage){
		this.ggFMTimer = new Timer(this,this.delayMessage,10,0);
		this.firstMessage = true;
	}
	return;
}
this.delayMessage = function()
{
	this.ship.commsMessage(expandDescription("[EVGG_FIRST_MESSAGE]"),player.ship);
	return;
}
this.playerWillEnterWitchspace = function()
{
	if(this.ggFMTimer){
		this.ggFMTimer.stop();
		delete this.ggFMTimer;
	}
}
